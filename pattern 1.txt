import java.util.Scanner;
class Main
{
    public static void main(String args[])
    {
        Scanner sc=new Scanner(System.in);
      int a=sc.nextInt();
      int b=(a*2)-1;
      for(int i=1;i<=a;i++)
      {
        for(int j=1;j<=b;j++)
        {
          System.out.print(j+" ");
        }
           b=b-2;
        System.out.println();
     
  }
}    
}
