import java.util.Scanner;
class Main
{
 public static void main(String args[]) 
 {
  Scanner sc=new Scanner(System.in);
   int a=sc.nextInt();
   char s='*';
   for(int i=0;i<a;i++)
   {
     for(int j=0;j<=i;j++)
     {
       System.out.print(s);
       if(s=='*')
       {
       s='#';
       }
       else
       {
       s='*';
       }
     }
      System.out.println("");
   }
 }
}
